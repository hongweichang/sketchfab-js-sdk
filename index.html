function login() {
    // Authenticate with OAuth2
    SketchfabSDK.connect()
        .then(function(grant){
            // Get user info
            return SketchfabSDK.Users.me(grant.access_token);
        })
        .then(function(response){
            // Log user info
            console.log(response);
        })
        .catch(function(error){
            console.log(error);
        });
}
